const CSV_URL = 'https://raw.githubusercontent.com/krocks9903/EagleGIS/script/estero_map_data_polished.csv'; let allData = []; // Helper to turn any weird Estero date string into a sortable number function getSortableTime(row) { const rawDate = row.ArcGIS_Date || row['Meeting Date'] || ""; const parsed = Date.parse(rawDate); // If standard parsing works, use it if (!isNaN(parsed)) return parsed; // Manual fallback: Extract 4-digit year if standard parsing fails const yearMatch = rawDate.match(/\d{4}/); if (yearMatch) { // Return the start of that year as a timestamp return new Date(yearMatch[0], 0, 1).getTime(); } return 0; // Absolute fallback for broken data } async function init() { console.log("Sniffing out the CSV... 🐾"); Papa.parse(CSV_URL, { download: true, header: true, skipEmptyLines: true, error: function(err) { console.error("Hiss! PapaParse failed:", err); document.getElementById('data-list').innerHTML = '
Failed to load data. Check connection.
'; }, complete: (results) => { console.log("Data caught! Sorting now..."); // Filter out truly empty rows allData = results.data.filter(row => row['Action Taken'] && row['Action Taken'].trim() !== ""); // ONE MASTER SORT: Most recent years first, then months allData.sort((a, b) => getSortableTime(b) - getSortableTime(a)); renderList(allData); setupSearch(); } }); } function renderList(data) { const list = document.getElementById('data-list'); list.innerHTML = ''; if (data.length === 0) { list.innerHTML = '
No matching events found. 🐾
'; return; } data.forEach(row => { const card = document.createElement('div'); card.className = "p-5 bg-slate-900 rounded-2xl border-2 border-slate-800 shadow-xl hover:border-indigo-500 transition-all cursor-pointer group mb-4"; card.innerHTML = `
${row['Meeting Date']}
"${row['Action Taken']}"
Expand Details 🐾 View PDF
`; card.onclick = (e) => { if (e.target.tagName === 'A') return; card.classList.toggle('expanded'); }; list.appendChild(card); }); } function setupSearch() { const input = document.getElementById('main-search'); input.addEventListener('input', (e) => { const term = e.target.value.toLowerCase().trim(); const filtered = allData.filter(row => { const action = (row['Action Taken'] || '').toLowerCase(); const date = (row['Meeting Date'] || '').toLowerCase(); return action.includes(term) || date.includes(term); }); renderList(filtered); }); } init();
